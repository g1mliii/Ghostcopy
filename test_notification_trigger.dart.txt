// TEST NOTIFICATION TRIGGER
// Add this method to SpotlightScreenState class to test notification actions
//
// Usage: Call this method from anywhere to trigger a test clickable notification
//
void _triggerTestNotification() {
  widget.notificationService?.showClickableToast(
    message: 'Test Clipboard Notification - Click to copy',
    actionLabel: 'Copy Test',
    duration: const Duration(seconds: 10),
    onAction: () async {
      // Test: Copy sample content to clipboard
      final testContent = 'Hello from GhostCopy! This is a test clipboard notification. Timestamp: ${DateTime.now()}';
      try {
        await ClipboardService.instance.writeText(testContent);
        debugPrint('[Test] Copied test content from notification: $testContent');
        widget.notificationService?.showToast(
          message: 'Test content copied! Check your clipboard.',
          type: NotificationType.success,
        );
      } catch (e) {
        debugPrint('[Test] Failed to copy: $e');
        widget.notificationService?.showToast(
          message: 'Copy failed: $e',
          type: NotificationType.error,
        );
      }
    },
  );
}
